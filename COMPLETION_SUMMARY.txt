🎯 DATA STRUCTURE (Ready for API Integration)
═══════════════════════════════════════════════════════════════════════════════

Each dashboard expects this data structure:

FACULTY:
  {
    id, name, email, role, progress_percentage (82), token_balance (41),
    loan_balance, department_id
    schedule: [{ time, subject, room, status }],
    activeCommitments: [{ title, tokens, dueDate, status }]
  }

HOD:
  {
    department_id, name, organization_id,
    stats: { totalMinted, withFaculty, reversed, belowThreshold },
    heatmap: [{ dept, progress, status }],
    approvals: [{ facultyName, tokens, status }],
    loans: [{ name, amount, monthsInDebt, status }]
  }

DIRECTOR:
  {
    organization_id, name,
    stats: { minted, held, belowThreshold, reversed },
    heatmap: [{ dept, progress, status }],
    loans: [{ name, dept, amount, status }],
    approvals: [{ name, dept, tokens, reason }]
  }

FINANCE:
  {
    stats: { readyForReversal, directorWallet, pendingSalary },
    batchReadiness: percentage,
    faculty: [{ name, id, dept, balance, status }],
    transactions: [{ type, timestamp, amount/batch }]
  }
